```sql
-- local_message ddl
CREATE TABLE `local_message` (
  `id` bigint(20) unsigned NOT NULL AUTO_INCREMENT COMMENT '主键ID',
  `create_time` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `update_time` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间（自动更新）',
  `deleted` tinyint(4) NOT NULL DEFAULT '0' COMMENT '逻辑删除标记（0：未删除；1：已删除）',
  `req_snapshot` text NOT NULL COMMENT '请求参数快照（JSON格式）',
  `status` varchar(16) NOT NULL DEFAULT 'INIT' COMMENT '状态（INIT：初始；FAIL：失败；SUCCESS：成功）',
  `biz_code` varchar(16) NOT NULL DEFAULT 'INIT' COMMENT '业务code',
  `next_retry_time` bigint(20) NOT NULL DEFAULT '0' COMMENT '下一次重试时间戳（毫秒）',
  `retry_times` int(11) NOT NULL DEFAULT '0' COMMENT '已重试次数',
  `max_retry_times` int(11) NOT NULL COMMENT '最大重试次数',
  `fail_reason` text COMMENT '失败原因描述',
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=4 DEFAULT CHARSET=utf8mb4 COMMENT='本地消息表（事务消息落地表）';



-- users ddl
CREATE TABLE `users` (
  `id` int(10) unsigned NOT NULL AUTO_INCREMENT COMMENT '用户唯一ID',
  `email` varchar(100) NOT NULL COMMENT '用户邮箱（登录账号）',
  `password` varchar(60) NOT NULL COMMENT '密码哈希（bcrypt/Argon2算法）',
  `name` varchar(50) NOT NULL COMMENT '用户姓名/昵称',
  `phone` varchar(20) DEFAULT NULL COMMENT '用户手机号',
  `image` varchar(255) DEFAULT NULL COMMENT '用户头像URL',
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '账号创建时间',
  `updated_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '账号最后更新时间',
  `last_login_at` datetime DEFAULT NULL COMMENT '用户最后登录时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `idx_email` (`email`) COMMENT '邮箱唯一索引（防止重复注册）',
  UNIQUE KEY `idx_phone` (`phone`) COMMENT '手机号唯一索引（防止重复绑定）',
  KEY `idx_created_at` (`created_at`) COMMENT '创建时间索引（优化按时间查询）'
) ENGINE=InnoDB AUTO_INCREMENT=11 DEFAULT CHARSET=utf8mb4 COMMENT='用户表';




-- ecommerce_orders ddl
CREATE TABLE `ecommerce_orders` (
  `order_id` bigint(20) NOT NULL AUTO_INCREMENT COMMENT '订单 ID，唯一标识',
  `customer_id` bigint(20) NOT NULL COMMENT '客户 ID，关联客户信息表',
  `order_date` datetime NOT NULL COMMENT '下单日期和时间',
  `shipping_date` datetime DEFAULT NULL COMMENT '发货日期',
  `delivery_date` datetime DEFAULT NULL COMMENT '预计交货日期',
  `product_id` varchar(255) COLLATE utf8mb4_bin NOT NULL COMMENT '产品 ID，关联产品信息表',
  `product_name` varchar(255) COLLATE utf8mb4_bin NOT NULL COMMENT '产品名称',
  `quantity` int(11) NOT NULL COMMENT '购买该产品的数量',
  `unit_price` decimal(10,2) NOT NULL COMMENT '单个产品的价格',
  `total_price` decimal(10,2) NOT NULL COMMENT '订单总金额',
  `payment_method` varchar(50) COLLATE utf8mb4_bin DEFAULT NULL COMMENT '支付方式(如信用卡、PayPal、转账等)',
  `shipping_address` varchar(255) COLLATE utf8mb4_bin DEFAULT NULL COMMENT '配送地址',
  `billing_address` varchar(255) COLLATE utf8mb4_bin DEFAULT NULL COMMENT '账单地址',
  `order_status` varchar(50) COLLATE utf8mb4_bin NOT NULL COMMENT '订单状态(如待支付、已支付、配送中、已完成、已取消等)',
  `tracking_number` varchar(100) COLLATE utf8mb4_bin DEFAULT NULL COMMENT '物流跟踪号',
  `courier_service` varchar(100) COLLATE utf8mb4_bin DEFAULT NULL COMMENT '快递服务商',
  `customer_notes` text COLLATE utf8mb4_bin COMMENT '客户留言备注',
  `internal_notes` text COLLATE utf8mb4_bin COMMENT '内部处理备注',
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '记录创建时间',
  PRIMARY KEY (`order_id`)
) ENGINE=InnoDB AUTO_INCREMENT=103 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_bin;


```
